        }

        System.out.print(" Enter another number: ") ;
        int another_number = ade.nextInt() ;

        int [] resized_array = new int [array.length + 1] ; // Updating array length and creating a new array
        for(int array_element_counter = 0; array_element_counter <= array.length - 1; array_element_counter++)
        {
        resized_array [array_element_counter] = array [array_element_counter] ;
        }
        resized_array[resized_array.length - 1] = another_number ;

        System.out.println(" After resizing the array: ") ;
        for(int array_element_counter = 0; array_element_counter <= resized_array.length - 1; array_element_counter++)
        {
        System.out.print(resized_array [array_element_counter] + " ") ;
    }
    }

import java.util.Scanner;

public class task_3_A_B
{
    public static void main(String [] args)
    {
        Scanner ade = new Scanner(System.in) ;
        
        System.out.print(" N = ") ; // Array Length
        int array_length = ade.nextInt() ;
        int [] array = new int [array_length] ; // Creating array of given length

        for(int array_element_counter = 0; array_element_counter <= array.length - 1; array_element_counter++)
        {
        System.out.print(" Enter a number: ") ;
        array [array_element_counter] = ade.nextInt() ;
        }

        // a) Out of place (Creating a new array)
        int [] new_array_a = new int [array_length] ;
        for(int array_element_counter = 0; array_element_counter <= array.length - 1; array_element_counter++)
        {
        new_array_a [array_element_counter] = array [array_length - 1 - array_element_counter] ;
        }

        System.out.println(" Reversed using a new array: ") ;
        for(int array_element_counter = 0; array_element_counter <= array.length - 1; array_element_counter++)
        {
        System.out.print(new_array_a [array_element_counter] + " ") ;
        }
        System.out.println() ; // Next line

        // b) In place (Within same array)
        for(int array_element_counter = 0; array_element_counter <= array.length / 2; array_element_counter++)
        {
        int temp = array [array_element_counter] ;
        array [array_element_counter] = array [array_length - 1 - array_element_counter] ;
        array [array_length - 1 - array_element_counter] = temp ;
        }

        System.out.println(" Reversed the original array: ") ;
        for(int array_element_counter = 0; array_element_counter <= array.length - 1; array_element_counter++)
        {
        System.out.print(array [array_element_counter] + " ") ;
        }
    }
}

        double [] array = new double [array_length] ; // Creating array of given length

        for(int array_element_counter = 0; array_element_counter <= array.length - 1; array_element_counter++)
        {
        System.out.print(" Enter a number: ") ;
        array [array_element_counter] = ade.nextDouble() ;
        }

        double max_element = array [0] ;
        int max_index = 0 ;
        double min_element = array [0] ;
        int min_index = 0 ;
        double sum = 0 ;

        for(int array_element_counter = 0; array_element_counter <= array.length - 1; array_element_counter++)
        {
        sum += array [array_element_counter] ;

                if(array [array_element_counter] > max_element)
                {
                max_element = array [array_element_counter] ;
                max_index = array_element_counter ;
                }

                if(array [array_element_counter] < min_element)
                {
                min_element = array [array_element_counter] ;
                min_index = array_element_counter ;
                }
        }
        double average = sum / array.length ;

        System.out.println(" Maximum element " + max_element + " found at index " + max_index) ;

import java.util.Scanner;
public class task_10
{
    public static void main(String [] args)
    {
        Scanner ade = new Scanner(System.in) ;

        System.out.print(" N = ") ; // Array Length
        int array_length = ade.nextInt() ;
        int [] marks = new int [array_length] ; // Creating array of given length for marks
        String [] names = new String [array_length] ; // Creating array of given length for names

        for(int array_element_counter = 0; array_element_counter <= array_length - 1; array_element_counter++)
        {
        System.out.print(" Enter marks: ") ;
        marks [array_element_counter] = ade.nextInt() ;
        ade.nextLine() ;
        System.out.print(" Enter names: ") ;
        names [array_element_counter] = ade.nextLine() ;
        }

        for(int array_element_counter = 0; array_element_counter <= (marks.length - 1) - 1; array_element_counter++)
        {
            for(int current_array_element_counter = 0; current_array_element_counter <= (marks.length - 1) - array_element_counter - 1; current_array_element_counter++)
            {
                if(marks [current_array_element_counter] > marks [current_array_element_counter + 1])
                {
                // Swapping marks
                int temp_marks = marks [current_array_element_counter] ;
                marks [current_array_element_counter] = marks [current_array_element_counter + 1] ;
                marks [current_array_element_counter + 1] = temp_marks ;

                // Swapping corresponding names
                String temp_names = names [current_array_element_counter] ;
                names [current_array_element_counter] = names [current_array_element_counter + 1] ;
                names [current_array_element_counter + 1] = temp_names ;
                }
            }
        }

        System.out.println(" Sorted Array: ") ;
        for(int array_element_counter = 0; array_element_counter <= array_length - 1; array_element_counter++)
        {
        System.out.print(marks [array_element_counter] + " ");
        }
        System.out.println() ; // Next line

        for(int array_element_counter = 0; array_element_counter <= array_length - 1; array_element_counter++)
        {
        System.out.print(names [array_element_counter] + " ") ;
        }
    }
}
