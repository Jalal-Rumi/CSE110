public class task_1
{
public static void main (String [] args)
    {
    Scanner ade = new Scanner(System.in) ;

    System.out.println(" You entered " + sample_input_number) ; // You entered 5
    int new_input ;
    double avg = 0 ;
    double sum = 0 ;
    int even_number_counter = 0 ;
    int max_even_positive_number = 0 ;
    int min_even_positive_number = 0 ;

    for(int counter = 1; counter <= sample_input_number; counter++)
    {
    System.out.print(" Enter a value = ") ;
    new_input = ade.nextInt() ;
    System.out.println(" You entered a input : " + new_input) ; // 12, -8, 19, 8, -1

        if(new_input > 0 && new_input % 2 == 0)
        {
        even_number_counter ++ ;
            if(even_number_counter >= 1)
            {
            sum += new_input ;
            avg = sum / even_number_counter ;
            }
            if(new_input >= max_even_positive_number)
            {
            max_even_positive_number = new_input ;
            }
            if(new_input < max_even_positive_number)
            {
            min_even_positive_number = new_input ;
            }
        }
    }
    System.out.println(" Average = " + avg + " ") ;
    System.out.println(" Max = " + max_even_positive_number) ;
    System.out.println(" Min = " + min_even_positive_number) ;
    ade.close() ;
    }
}

        int counter = 1 ;
        int starting_digit = 2 ;
        int flag = 0 ;

        if(n > 0)
        {
            while(counter <= n)
            {
    
                for(int starting_divisor = 2; starting_divisor < starting_digit; starting_divisor ++)
                {
                    if(starting_digit % starting_divisor == 0)
                    {
                    flag = 1 ;
                    }
                }
            if(flag == 0)
            {
            System.out.println(starting_digit + " which is a prime number ") ;
            counter ++ ;

            flag = 0 ; // Resetting flag for each number
            }
        }

        else
        {
            System.out.println(" Invalid ") ;
        }
    ade.close() ;

        for(int loop_counter = 1; loop_counter <= n; loop_counter ++)

            for(int part_counter = 1; part_counter <= loop_counter; part_counter ++)

            individual_part_sum = individual_part_sum + part_counter ;

        individual_part_sum = 0 ; // resetting it for each part as each part starts with 1

    int sum_of_odd_numbers = 0 ;

    for(int counter = 1; counter <= n; counter ++)
    {

    // Creating (X, Y) pair for next case
    System.out.println(" Enter X = ") ;
    int first_digit = ade.nextInt() ; // X
    System.out.println(" Enter Y = ") ;
    int second_digit = ade.nextInt() ; // Y


        while(digit_counter <= second_digit)
        {

            if(first_digit % 2 != 0)
            {
            sum_of_odd_numbers = sum_of_odd_numbers + first_digit ;

            }
        first_digit ++ ;
        }

{
public static void main (String [] args)
    {
    Scanner ade = new Scanner(System.in) ;
    System.out.print(" Starting Value = ") ;
    int start = ade.nextInt() ;
    System.out.print(" Ending Value = ") ;
    int end = ade.nextInt() ;
    int sum = 0 ;
    System.out.println(" Armstrong Numbers : ") ;

    for(int tempo = start; tempo <= end; tempo ++) // 371
    {

        for(int calculator = tempo; calculator > 0;  )
        {
        int each_digit_reverse = calculator % 10 ; // 1, 7, 3
        sum = sum + (each_digit_reverse * each_digit_reverse * each_digit_reverse) ;
        calculator = calculator / 10 ; // 371, 37, 3
        }

    if (tempo == sum)
    {
    System.out.println(tempo) ;
    }
    sum = 0 ; // Resetting for next number
    }
    ade.close() ;
    }
}
