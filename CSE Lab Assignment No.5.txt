{
public static void main (String [] args)
    {
    Scanner ade = new Scanner(System.in) ;
    System.out.print(" Please enter a sample string input = ") ;
    String sample_string_input = ade.nextLine() ;
    int vowel_counter = 0 ;
    int consonant_counter = 0 ;


    for(int string_position = 0; string_position <= sample_string_input.length() - 1; string_position++)
    {
    char current_character_position = sample_string_input.charAt(string_position) ;

        if(current_character_position == 'a')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'e')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'i')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'o')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'u')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'A')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'E')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'I')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'O')
        {
        vowel_counter ++ ;
        continue ;
        }

        if(current_character_position == 'U')
        {
        vowel_counter ++ ;
        continue ;
        }

        else if(('a' <= current_character_position && current_character_position <= 'z') || ('A' <= current_character_position && current_character_position <= 'Z'))
        {
        consonant_counter ++ ;
        }
    }

    if((vowel_counter > 0 && vowel_counter % 3 == 0) && (consonant_counter > 0 && consonant_counter % 5 == 0))
    {
    System.out.println("Aaarr! Me Plunder!!") ;
    }

    else
    {
    System.out.println("Blimey! No Plunder!!") ;
    }
    }

import java.util.Scanner ;

public class task_8
{
public static void main (String [] args)
    {
    Scanner ade = new Scanner(System.in) ;
    System.out.print(" Please enter a sample string = ") ;
    String sample_string= ade.nextLine() ;
    String full_string = "" ;
    boolean flag = false ;

    for(int string_position = 0; string_position <= sample_string.length() - 1; string_position++)
    {
    char current_character_position = sample_string.charAt(string_position) ;

        if(('a' <= current_character_position && current_character_position <= 'z') || ('A' <= current_character_position && current_character_position <= 'Z'))
        {

            if(flag == false) // If flag is false, making the letter lowercase, otherwise making it uppercase
            {
            full_string += Character.toLowerCase(current_character_position) ;
            flag = true ; // for next character
            }

            else
            {
            full_string += Character.toUpperCase(current_character_position) ;
            flag = false ; // for next character
            }
        }

        else
        {
        full_string += current_character_position ;
        }
    }
    System.out.print(full_string) ;
    }
}
